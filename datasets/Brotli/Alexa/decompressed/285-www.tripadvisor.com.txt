//one place to hold logic for all versions of the test
function _updateValues(valuesObj, show, notify) {
var nRooms = valuesObj.nRooms || _numRooms
, nAdults = valuesObj.nAdults || _numAdults
, nChildren = (typeof(valuesObj.nChildren) === 'undefined' ? _numChildren : valuesObj.nChildren)
, newNumAdults
, newNumChildren
, newNumRooms;
if (_numRooms === nRooms && _numAdults === nAdults && _numChildren === nChildren) {
return false;
}
if (nRooms !== _numRooms) {
newNumRooms = nRooms;
if (newNumRooms > nAdults) {
//at least one adult per room
newNumAdults = newNumRooms;
} else {
//no more then 4 adults per room
newNumAdults = Math.min(nRooms * MAX_ADULTS_PER_ROOM, nAdults);
}
//no more then 8 children per room.
newNumChildren = Math.min(nRooms * MAX_CHILDREN_PER_ROOM, nChildren);
} else {
//minimal number of rooms as dictated by number of adults
var adultsRequiredNumberOfRooms = Math.ceil(nAdults / MAX_ADULTS_PER_ROOM);
//but they might want more rooms
adultsRequiredNumberOfRooms = Math.max(adultsRequiredNumberOfRooms, nRooms);
//minimal number of rooms as dictated by number of children
var childrenRequiredNumberOfRooms = Math.ceil(nChildren / MAX_CHILDREN_PER_ROOM);
//maximal number of rooms required
newNumRooms = Math.max(adultsRequiredNumberOfRooms, childrenRequiredNumberOfRooms);
//at least one adult per room
newNumRooms = Math.min(newNumRooms, nAdults);
//force max children per room
newNumChildren = Math.min(nChildren, newNumRooms * MAX_CHILDREN_PER_ROOM);
newNumAdults = nAdults;
}
var changed = false;
if (newNumRooms !== _numRooms) {
_setRoomsLabel(newNumRooms);
changed = true;
}
if (newNumChildren !== _numChildren) {
_createAgePickers(newNumChildren, show);
}
if (newNumAdults !== _numAdults) {
_setAdultsLabel(newNumAdults);
changed = true;
}
if (newNumChildren !== _numChildren) {
_setChildrenLabel(newNumChildren);
changed = true;
}
_numRooms = newNumRooms;
_numAdults = newNumAdults;
_numChildren = newNumChildren;
if (notify) {
_updateGuestsRoomsObject(false);
}
return changed;
}

//updates the global object with selected values
function _updateGuestsRoomsObject() {

rooms:_numRooms,

};

if(!changed )
{

}

if (changed) {
ta.store('guests.rooms.object', obj);

//verify fields values
// @params interactive - indicated if should show visuals on errors
// @returns - boolean indicating if error was found
function _verifyFields(interactive, agesOnly, removeAgesErrorIsPossible) {
var hasErrors = false,
hasAgesError = false;
if (!_calendarProxy.hasDates()) {
if (interactive) {
_calendarProxy.showCalendar();
}
hasErrors = true;
}
var ages = _gatherChildrenAges();
for (var i = 0 ; i < ages.length ; ++i) {
if (ages[i] < 0) {
hasErrors = true;
hasAgesError = true;
if (interactive || agesOnly) {
api.addClass(api.find('.age-picker-' + (i + 1) + ' .picker-inner', _agesPickerContainerElement), 'error');
}
}
}
if (interactive || agesOnly) {
_updateAgesErrorMsg(false, hasAgesError);
}
if (removeAgesErrorIsPossible && !hasAgesError) {
api.addClass(api.find('.ages-error', _agesWrapElement), 'hidden');
}
return !hasErrors;
}
function _agesPopupVerify() {

var ages = _gatherChildrenAges();
for (var i = 0 ; i < ages.length ; ++i) {

hasAgesError = true;
break;
}
}
if (!hasAgesError) {
api.removeClass(_popupUpdateButton, 'disabled');
} else {
api.addClass(_popupUpdateButton, 'disabled');
}
return !hasAgesError;
}

// somewhat similar to the one in Desktop.js, but can get which picker type
function _findClosestDatepicker(type) {
var el,
target = _widgetDiv;
while (target && target !== document.body) {
el = target.getElement('[data-dateType='+type+']');
if (el) {
return el;
}
target = target.parentNode;
}
return null;
}
function _findSearchButton()
{
// looking for the relevant search button - it is located outside of the widget
